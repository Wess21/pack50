### Task 2: Implement Authentication Flow
**Objective**: Login page with JWT token storage

**Steps**:
1. Create `admin/src/api/auth.ts`:
   ```typescript
   import axios from 'axios';

   const API_BASE = '/api/admin';

   export interface LoginResponse {
     token: string;
     username: string;
   }

   export async function login(username: string, password: string): Promise<LoginResponse> {
     const response = await axios.post(`${API_BASE}/login`, { username, password });
     return response.data;
   }

   export function getToken(): string | null {
     return localStorage.getItem('admin_token');
   }

   export function setToken(token: string): void {
     localStorage.setItem('admin_token', token);
   }

   export function clearToken(): void {
     localStorage.removeItem('admin_token');
   }

   export function getAuthHeader() {
     const token = getToken();
     return token ? { Authorization: `Bearer ${token}` } : {};
   }
   ```
2. Create `admin/src/pages/Login.tsx`:
   ```tsx
   import React, { useState } from 'react';
   import { useNavigate } from 'react-router-dom';
   import { login, setToken } from '../api/auth';

   export default function Login() {
     const [username, setUsername] = useState('');
     const [password, setPassword] = useState('');
     const [error, setError] = useState('');
     const [loading, setLoading] = useState(false);
     const navigate = useNavigate();

     const handleSubmit = async (e: React.FormEvent) => {
       e.preventDefault();
       setError('');
       setLoading(true);

       try {
         const response = await login(username, password);
         setToken(response.token);
         navigate('/admin/dashboard');
       } catch (err: any) {
         setError(err.response?.data?.error || 'Ошибка входа');
       } finally {
         setLoading(false);
       }
     };

     return (
       <div className="min-h-screen bg-gray-100 flex items-center justify-center">
         <div className="bg-white p-8 rounded-lg shadow-md w-96">
           <h1 className="text-2xl font-bold mb-6">Вход в админ-панель</h1>
           <form onSubmit={handleSubmit}>
             <div className="mb-4">
               <label className="block text-sm font-medium mb-2">Логин</label>
               <input
                 type="text"
                 value={username}
                 onChange={(e) => setUsername(e.target.value)}
                 className="w-full px-3 py-2 border rounded"
                 required
               />
             </div>
             <div className="mb-6">
               <label className="block text-sm font-medium mb-2">Пароль</label>
               <input
                 type="password"
                 value={password}
                 onChange={(e) => setPassword(e.target.value)}
                 className="w-full px-3 py-2 border rounded"
                 required
               />
             </div>
             {error && <p className="text-red-600 text-sm mb-4">{error}</p>}
             <button
               type="submit"
               disabled={loading}
               className="w-full bg-blue-600 text-white py-2 rounded hover:bg-blue-700 disabled:opacity-50"
             >
               {loading ? 'Вход...' : 'Войти'}
             </button>
           </form>
         </div>
       </div>
     );
   }
   ```
3. Create protected route wrapper `admin/src/components/ProtectedRoute.tsx`:
   ```tsx
   import React from 'react';
   import { Navigate } from 'react-router-dom';
   import { getToken } from '../api/auth';

   export default function ProtectedRoute({ children }: { children: React.ReactNode }) {
     const token = getToken();

     if (!token) {
       return <Navigate to="/admin/login" replace />;
     }

     return <>{children}</>;
   }
   ```

### Task 3: Build Configuration Page (Model + API Keys)
**Objective**: UI for selecting AI model and managing API keys

**Steps**:
1. Create `admin/src/api/config.ts`:
   ```typescript
   import axios from 'axios';
   import { getAuthHeader } from './auth';

   const API_BASE = '/api/admin';

   export interface BotConfig {
     active_model: string;
     active_template: string;
     webhook_url?: string;
   }

   export async function getConfig(): Promise<BotConfig> {
     const response = await axios.get(`${API_BASE}/config`, { headers: getAuthHeader() });
     return response.data;
   }

   export async function updateConfig(updates: Partial<BotConfig> & {
     anthropic_api_key?: string;
     openai_api_key?: string;
   }): Promise<void> {
     await axios.put(`${API_BASE}/config`, updates, { headers: getAuthHeader() });
   }

   export async function testModel(provider: string, apiKey: string): Promise<boolean> {
     try {
       const response = await axios.post(
         `${API_BASE}/test-model`,
         { provider, api_key: apiKey },
         { headers: getAuthHeader() }
       );
       return response.data.success;
     } catch {
       return false;
     }
   }
   ```

### Task 4: Build Prompts Page (Template Editor)
**Objective**: Select and edit system prompt templates

**Steps**:
1. Create `admin/src/api/prompts.ts`:
   ```typescript
   import axios from 'axios';
   import { getAuthHeader } from './auth';

   const API_BASE = '/api/admin';

   export interface Prompt {
     template_name: string;
     content: string;
   }

   export async function getPrompts(): Promise<Prompt[]> {
     const response = await axios.get(`${API_BASE}/prompts`, { headers: getAuthHeader() });
     return response.data;
   }

   export async function updatePrompt(templateName: string, content: string): Promise<void> {
     await axios.put(`${API_BASE}/prompts/${templateName}`, { content }, { headers: getAuthHeader() });
   }
   ```
2. Add to `src/api/routes/admin.ts` (backend):
   ```typescript
   // Get all prompts (custom + defaults)
   router.get('/prompts', requireAdmin, async (req, res) => {
     const customs = await pool.query('SELECT template_name, content FROM system_prompts');

     // Merge with defaults from system-prompts.ts
     const defaults = ['consultant', 'support', 'orderTaker'];
     const prompts = defaults.map(name => {
       const custom = customs.rows.find(r => r.template_name === name);
       return custom || { template_name: name, content: `[Default ${name} prompt]` };
     });

     res.json(prompts);
   });

   // Update prompt
   router.put('/prompts/:name', requireAdmin, async (req: AdminRequest, res) => {
     const { name } = req.params;
     const { content } = req.body;

     await pool.query(
       `INSERT INTO system_prompts (template_name, content) VALUES ($1, $2)
        ON CONFLICT (template_name) DO UPDATE SET content = $2, updated_at = NOW()`,
       [name, content]
     );

     logger.info('System prompt updated', { template: name, adminId: req.adminId });
     res.json({ success: true });
   });
   ```
3. Create `admin/src/pages/Prompts.tsx`:
   ```tsx
   import React, { useState, useEffect } from 'react';
   import { getConfig, updateConfig } from '../api/config';
   import { getPrompts, updatePrompt, Prompt } from '../api/prompts';

   export default function Prompts() {
     const [prompts, setPrompts] = useState<Prompt[]>([]);
     const [activeTemplate, setActiveTemplate] = useState('');
     const [editContent, setEditContent] = useState('');
     const [saving, setSaving] = useState(false);
     const [message, setMessage] = useState('');

     useEffect(() => {
       loadData();
     }, []);

     const loadData = async () => {
       const config = await getConfig();
       const promptsList = await getPrompts();
       setActiveTemplate(config.active_template);
       setPrompts(promptsList);

       const current = promptsList.find(p => p.template_name === config.active_template);
       if (current) setEditContent(current.content);
     };

     const handleTemplateChange = async (templateName: string) => {
       setActiveTemplate(templateName);
       await updateConfig({ active_template: templateName });

       const prompt = prompts.find(p => p.template_name === templateName);
       if (prompt) setEditContent(prompt.content);
     };

     const handleSave = async () => {
       setSaving(true);
       try {
         await updatePrompt(activeTemplate, editContent);
         setMessage('✅ Промпт сохранен');
       } catch (err) {
         setMessage('❌ Ошибка сохранения');
       } finally {
         setSaving(false);
       }
     };

     return (
       <div className="p-6">
         <h1 className="text-3xl font-bold mb-6">Системные промпты</h1>

         <div className="bg-white p-6 rounded-lg shadow">
           <h2 className="text-xl font-semibold mb-4">Активный шаблон</h2>
           <select
             value={activeTemplate}
             onChange={(e) => handleTemplateChange(e.target.value)}
             className="w-full px-3 py-2 border rounded mb-6"
           >
             {prompts.map(p => (
               <option key={p.template_name} value={p.template_name}>
                 {p.template_name}
               </option>
             ))}
           </select>

           <h2 className="text-xl font-semibold mb-4">Редактор промпта</h2>
           <textarea
             value={editContent}
             onChange={(e) => setEditContent(e.target.value)}
             rows={15}
             className="w-full px-3 py-2 border rounded font-mono text-sm mb-4"
           />

           <p className="text-sm text-gray-600 mb-4">
             Доступные переменные: {'{business_name}'}, {'{current_date}'}
           </p>

           {message && <p className="mb-4 text-sm">{message}</p>}

           <button
             onClick={handleSave}
             disabled={saving}
             className="w-full bg-blue-600 text-white py-2 rounded hover:bg-blue-700 disabled:opacity-50"
           >
             {saving ? 'Сохранение...' : 'Сохранить промпт'}
           </button>
         </div>
       </div>
     );
   }
   ```

**Verification**:
- Prompts page loads available templates
- Template dropdown changes active selection
- Textarea loads current prompt content
- Save button persists changes to database
