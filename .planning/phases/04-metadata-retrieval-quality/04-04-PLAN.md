### Task 1: Add Analytics Event Tracking to Message Handler
**Objective**: Track conversation events in analytics_events table

**Steps**:
1. Create `src/services/analytics.ts`:
   ```typescript
   import { pool } from '../db/pool.js';
   import { logger } from '../utils/logger.js';

   export type EventType =
     | 'conversation_start'
     | 'message_sent'
     | 'webhook_triggered'
     | 'llm_response'
     | 'rag_search'
     | 'error_occurred';

   export interface AnalyticsEvent {
     event_type: EventType;
     user_id: number;
     metadata?: Record<string, any>;
   }

   export class AnalyticsService {
     async track(event: AnalyticsEvent): Promise<void> {
       try {
         await pool.query(
           'INSERT INTO analytics_events (event_type, user_id, metadata) VALUES ($1, $2, $3)',
           [event.event_type, event.user_id, JSON.stringify(event.metadata || {})]
         );
       } catch (error) {
         // Don't fail requests if analytics fails
         logger.error('Analytics tracking failed', { error, event });
       }
     }

     async trackConversationStart(userId: number): Promise<void> {
       await this.track({ event_type: 'conversation_start', user_id: userId });
     }

     async trackMessage(userId: number, metadata: {
       message_length: number;
       response_time_ms: number;
       model_used?: string;
     }): Promise<void> {
       await this.track({
         event_type: 'message_sent',
         user_id: userId,
         metadata,
       });
     }

     async trackWebhook(userId: number, success: boolean): Promise<void> {
       await this.track({
         event_type: 'webhook_triggered',
         user_id: userId,
         metadata: { success },
       });
     }

     async trackLLMResponse(userId: number, metadata: {
       input_tokens: number;
       output_tokens: number;
       model: string;
     }): Promise<void> {
       await this.track({
         event_type: 'llm_response',
         user_id: userId,
         metadata,
       });
     }

     async trackRAGSearch(userId: number, metadata: {
       query_length: number;
       results_count: number;
       search_time_ms: number;
     }): Promise<void> {
       await this.track({
         event_type: 'rag_search',
         user_id: userId,
         metadata,
       });
     }
   }

   export const analyticsService = new AnalyticsService();
   ```
2. Integrate into `src/bot/handlers/message-handler.ts`:

### Task 2: Build Analytics Aggregation Queries
**Objective**: Create API endpoint for dashboard metrics

**Steps**:
1. Create `src/services/analytics-queries.ts`:
   ```typescript
   import { pool } from '../db/pool.js';

   export interface DashboardMetrics {
     totalConversations: number;
     totalMessages: number;
     avgResponseTimeMs: number;
     totalWebhooks: number;
     uniqueUsers: number;
     tokenUsage: {
       inputTokens: number;
       outputTokens: number;
     };
     dailyActivity: Array<{ date: string; count: number }>;
   }

   export async function getDashboardMetrics(
     startDate: Date,
     endDate: Date
   ): Promise<DashboardMetrics> {
     // Total conversations
     const conversationsResult = await pool.query(
       `SELECT COUNT(*) as count FROM analytics_events
        WHERE event_type = 'conversation_start'
        AND timestamp >= $1 AND timestamp <= $2`,
       [startDate, endDate]
     );

     // Total messages
     const messagesResult = await pool.query(
       `SELECT COUNT(*) as count, AVG((metadata->>'response_time_ms')::int) as avg_time
        FROM analytics_events
        WHERE event_type = 'message_sent'
        AND timestamp >= $1 AND timestamp <= $2`,
       [startDate, endDate]
     );

     // Unique users
     const usersResult = await pool.query(
       `SELECT COUNT(DISTINCT user_id) as count FROM analytics_events
        WHERE timestamp >= $1 AND timestamp <= $2`,
       [startDate, endDate]
     );

     // Webhook count
     const webhooksResult = await pool.query(
       `SELECT COUNT(*) as count FROM analytics_events
        WHERE event_type = 'webhook_triggered'
        AND timestamp >= $1 AND timestamp <= $2`,
       [startDate, endDate]
     );

     // Token usage
     const tokensResult = await pool.query(
       `SELECT
          SUM((metadata->>'input_tokens')::int) as input_tokens,
          SUM((metadata->>'output_tokens')::int) as output_tokens
        FROM analytics_events
        WHERE event_type = 'llm_response'
        AND timestamp >= $1 AND timestamp <= $2`,
       [startDate, endDate]
     );

     // Daily activity
     const dailyResult = await pool.query(
       `SELECT DATE(timestamp) as date, COUNT(*) as count
        FROM analytics_events
        WHERE event_type = 'message_sent'
        AND timestamp >= $1 AND timestamp <= $2
        GROUP BY DATE(timestamp)
        ORDER BY date DESC
        LIMIT 30`,
       [startDate, endDate]
     );

     return {
       totalConversations: parseInt(conversationsResult.rows[0]?.count || '0'),
       totalMessages: parseInt(messagesResult.rows[0]?.count || '0'),
       avgResponseTimeMs: parseInt(messagesResult.rows[0]?.avg_time || '0'),
       totalWebhooks: parseInt(webhooksResult.rows[0]?.count || '0'),
       uniqueUsers: parseInt(usersResult.rows[0]?.count || '0'),
       tokenUsage: {
         inputTokens: parseInt(tokensResult.rows[0]?.input_tokens || '0'),
         outputTokens: parseInt(tokensResult.rows[0]?.output_tokens || '0'),
       },
       dailyActivity: dailyResult.rows.map(row => ({
         date: row.date,
         count: parseInt(row.count),
       })),
     };
   }
   ```
2. Add to `src/api/routes/admin.ts`:
   ```typescript
   import { getDashboardMetrics } from '../../services/analytics-queries.js';

   // Get analytics metrics
   router.get('/analytics', requireAdmin, async (req, res) => {
     const { days = 30 } = req.query;

     const endDate = new Date();
     const startDate = new Date();
     startDate.setDate(startDate.getDate() - parseInt(days as string));

     const metrics = await getDashboardMetrics(startDate, endDate);
     res.json(metrics);
   });
   ```

**Verification**:
- GET /api/admin/analytics returns metrics object
- totalConversations matches conversation_start event count
- avgResponseTimeMs calculates correctly
- dailyActivity returns last 30 days

### Task 3: Enhance RAG Retrieval with Metadata Filtering
**Objective**: Add optional metadata filters to vector search

**Steps**:
1. Update `src/services/retrieval.ts`:
   ```typescript
   export interface SearchFilters {
     doc_types?: string[]; // Filter by document type (pdf, docx, url)
     date_from?: Date;     // Filter by created_at >= date
     date_to?: Date;       // Filter by created_at <= date
     sources?: string[];   // Filter by specific source URLs/filenames
   }

   export class RetrievalService {
     async search(
       query: string,
       topK: number = 5,
       filters?: SearchFilters
     ): Promise<SearchResult[]> {
       const queryEmbedding = await this.embeddingService.embed(query);

       // Build WHERE clause for filters
       let whereClause = 'WHERE 1=1';
       const params: any[] = [JSON.stringify(queryEmbedding), topK];
       let paramIndex = 3;

       if (filters?.doc_types && filters.doc_types.length > 0) {
         whereClause += ` AND doc_type = ANY($${paramIndex})`;
         params.push(filters.doc_types);
         paramIndex++;
       }

       if (filters?.date_from) {
         whereClause += ` AND created_at >= $${paramIndex}`;
         params.push(filters.date_from);
         paramIndex++;
       }

       if (filters?.date_to) {
         whereClause += ` AND created_at <= $${paramIndex}`;
         params.push(filters.date_to);
         paramIndex++;
       }

       if (filters?.sources && filters.sources.length > 0) {
         whereClause += ` AND source = ANY($${paramIndex})`;
         params.push(filters.sources);
         paramIndex++;
       }

       const result = await pool.query(
         `SELECT id, content, source, page, doc_type, created_at,
                 1 - (embedding <=> $1::vector) AS similarity
          FROM document_chunks
          ${whereClause}
          ORDER BY embedding <=> $1::vector
          LIMIT $2`,
         params
       );

       return result.rows;
     }
   }
   ```
2. Add filter configuration to bot_config table (optional):
   ```sql
   ALTER TABLE bot_config ADD COLUMN default_filters JSONB;
   ```
3. Update message handler to use filters (if configured):
   ```typescript

   const filters = configResult.rows[0]?.default_filters || {};

   const searchResults = await retrievalService.search(messageText, 5, filters);
   ```

**Verification**:
- search() with doc_types=['pdf'] only returns PDF chunks
- search() with date_from filters old documents
- search() without filters works as before (backward compatible)

### Task 5: Add Analytics Retention Policy (Cleanup Job)
**Objective**: Prevent analytics table from growing indefinitely

**Steps**:
1. Create `src/jobs/analytics-cleanup.ts`:
   ```typescript
   import { pool } from '../db/pool.js';
   import { logger } from '../utils/logger.js';

   const RETENTION_DAYS = 90; // Keep 90 days of detailed events

   export async function cleanupOldAnalytics(): Promise<void> {
     const cutoffDate = new Date();
     cutoffDate.setDate(cutoffDate.getDate() - RETENTION_DAYS);

     const result = await pool.query(
       'DELETE FROM analytics_events WHERE timestamp < $1',
       [cutoffDate]
     );

     logger.info('Analytics cleanup completed', {
       deletedRows: result.rowCount,
       cutoffDate,
     });
   }

   // Run daily at 3 AM
   export function scheduleAnalyticsCleanup(): void {
     const runCleanup = async () => {
       const now = new Date();
       if (now.getHours() === 3 && now.getMinutes() === 0) {
         await cleanupOldAnalytics();
       }
     };

     // Check every minute
     setInterval(runCleanup, 60 * 1000);
     logger.info('Analytics cleanup scheduled (daily at 3 AM)');
   }
   ```
2. Call in `src/index.ts`:
   ```typescript
   import { scheduleAnalyticsCleanup } from './jobs/analytics-cleanup.js';

   // After bot starts
   scheduleAnalyticsCleanup();
   ```

**Verification**:
- Old events (>90 days) are deleted
- Cleanup runs daily without manual intervention
- Logs show deletion count
