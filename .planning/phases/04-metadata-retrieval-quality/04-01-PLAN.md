### Task 2: Implement Encryption Utilities
**Objective**: Secure API key encryption/decryption with AES-256

**Steps**:
1. Create `src/utils/encryption.ts`:
   ```typescript
   import crypto from 'crypto';
   import { env } from '../config/env.js';

   const ALGORITHM = 'aes-256-cbc';
   const ENCRYPTION_KEY = env.ENCRYPTION_KEY; // 32 bytes from .env

   export function encryptApiKey(plaintext: string): { encrypted: string; iv: string } {
     const iv = crypto.randomBytes(16);
     const cipher = crypto.createCipheriv(ALGORITHM, Buffer.from(ENCRYPTION_KEY, 'hex'), iv);
     let encrypted = cipher.update(plaintext, 'utf8', 'hex');
     encrypted += cipher.final('hex');
     return { encrypted, iv: iv.toString('hex') };
   }

   export function decryptApiKey(encrypted: string, iv: string): string {
     const decipher = crypto.createDecipheriv(
       ALGORITHM,
       Buffer.from(ENCRYPTION_KEY, 'hex'),
       Buffer.from(iv, 'hex')
     );
     let decrypted = decipher.update(encrypted, 'hex', 'utf8');
     decrypted += decipher.final('utf8');
     return decrypted;
   }
   ```
2. Update `src/config/env.ts`:
   - Add `ENCRYPTION_KEY: z.string().length(64)` (32 bytes hex)
   - Add to `.env.example`: `ENCRYPTION_KEY=your-32-byte-hex-key-here-64-chars`
3. Generate default encryption key in setup docs

**Verification**:
- Encrypt + decrypt roundtrip returns original value
- Different IVs produce different ciphertexts for same plaintext
- Decryption with wrong IV fails gracefully

### Task 3: Create Admin Authentication Middleware
**Objective**: JWT-based admin authentication for API protection

**Steps**:
1. Install dependencies: `npm install bcrypt jsonwebtoken @types/bcrypt @types/jsonwebtoken`
2. Create `src/middleware/admin-auth.ts`:
   ```typescript
   import jwt from 'jsonwebtoken';
   import { Request, Response, NextFunction } from 'express';
   import { env } from '../config/env.js';

   const JWT_SECRET = env.JWT_SECRET;
   const TOKEN_EXPIRY = '7d';

   export interface AdminRequest extends Request {
     adminId?: number;
   }

   export function generateToken(adminId: number): string {
     return jwt.sign({ adminId }, JWT_SECRET, { expiresIn: TOKEN_EXPIRY });
   }

   export function requireAdmin(req: AdminRequest, res: Response, next: NextFunction) {
     const token = req.headers.authorization?.replace('Bearer ', '');
     if (!token) {
       return res.status(401).json({ error: 'No token provided' });
     }

     try {
       const decoded = jwt.verify(token, JWT_SECRET) as { adminId: number };
       req.adminId = decoded.adminId;
       next();
     } catch (error) {
       return res.status(401).json({ error: 'Invalid or expired token' });
     }
   }
   ```
3. Update `src/config/env.ts`: Add `JWT_SECRET: z.string().min(32)`
4. Add to `.env.example`: `JWT_SECRET=your-jwt-secret-min-32-chars`

**Verification**:
- Token generation works and includes adminId
- requireAdmin middleware blocks requests without token
- requireAdmin middleware allows requests with valid token
- Expired tokens are rejected

### Task 4: Implement Configuration API Endpoints
**Objective**: CRUD operations for bot_config with secure key handling

**Steps**:
1. Create `src/api/routes/admin.ts`:
   ```typescript
   import { Router } from 'express';
   import bcrypt from 'bcrypt';
   import { pool } from '../../db/pool.js';
   import { requireAdmin, generateToken, AdminRequest } from '../../middleware/admin-auth.js';
   import { encryptApiKey, decryptApiKey } from '../../utils/encryption.js';
   import { logger } from '../../utils/logger.js';

   const router = Router();

   // Public: Admin login
   router.post('/login', async (req, res) => {
     const { username, password } = req.body;

     const result = await pool.query(
       'SELECT id, password_hash FROM admin_users WHERE username = $1',
       [username]
     );

     if (result.rows.length === 0) {
       return res.status(401).json({ error: 'Invalid credentials' });
     }

     const admin = result.rows[0];
     const valid = await bcrypt.compare(password, admin.password_hash);

     if (!valid) {
       return res.status(401).json({ error: 'Invalid credentials' });
     }

     const token = generateToken(admin.id);
     res.json({ token, username });
   });

   // Protected: Get bot configuration
   router.get('/config', requireAdmin, async (req, res) => {
     const result = await pool.query(
       'SELECT active_model, active_template, webhook_url FROM bot_config WHERE id = 1'
     );

     res.json(result.rows[0] || {});
   });

   // Protected: Update bot configuration
   router.put('/config', requireAdmin, async (req: AdminRequest, res) => {
     const { active_model, active_template, webhook_url, anthropic_api_key, openai_api_key } = req.body;

     const updates: string[] = [];
     const values: any[] = [];
     let paramIndex = 1;

     if (active_model) {
       updates.push(`active_model = $${paramIndex++}`);
       values.push(active_model);
     }

     if (active_template) {
       updates.push(`active_template = $${paramIndex++}`);
       values.push(active_template);
     }

     if (webhook_url !== undefined) {
       updates.push(`webhook_url = $${paramIndex++}`);
       values.push(webhook_url);
     }

     // Encrypt API keys if provided
     if (anthropic_api_key) {
       const { encrypted, iv } = encryptApiKey(anthropic_api_key);
       updates.push(`anthropic_api_key_encrypted = $${paramIndex++}`, `encryption_iv = $${paramIndex++}`);
       values.push(encrypted, iv);
     }

     if (openai_api_key) {
       const { encrypted, iv } = encryptApiKey(openai_api_key);
       updates.push(`openai_api_key_encrypted = $${paramIndex++}`, `encryption_iv = $${paramIndex++}`);
       values.push(encrypted, iv);
     }

     if (updates.length === 0) {
       return res.status(400).json({ error: 'No valid fields to update' });
     }

     updates.push(`updated_at = NOW()`);

     await pool.query(
       `UPDATE bot_config SET ${updates.join(', ')} WHERE id = 1`,
       values
     );

     logger.info('Bot configuration updated', { adminId: req.adminId });
     res.json({ success: true });
   });

   export default router;
   ```
2. Mount router in `src/api/index.ts`:
   ```typescript
   import adminRoutes from './routes/admin.js';
   app.use('/api/admin', adminRoutes);
   ```

**Verification**:
- POST /api/admin/login with valid credentials returns JWT token
- GET /api/admin/config without token returns 401
- GET /api/admin/config with token returns current config
- PUT /api/admin/config updates database and encrypts keys
- API keys never appear in response JSON

### Task 5: Seed Initial Admin User
**Objective**: Create default admin account for first-time setup

**Steps**:
1. Create `src/db/seed-admin.ts`:
   ```typescript
   import bcrypt from 'bcrypt';
   import { pool } from './pool.js';
   import { logger } from '../utils/logger.js';

   export async function seedAdminUser() {
     const defaultUsername = 'admin';
     const defaultPassword = process.env.DEFAULT_ADMIN_PASSWORD || 'changeme';

     // Check if admin exists
     const existing = await pool.query(
       'SELECT id FROM admin_users WHERE username = $1',
       [defaultUsername]
     );

     if (existing.rows.length > 0) {
       logger.info('Admin user already exists, skipping seed');
       return;
     }

     // Create admin
     const passwordHash = await bcrypt.hash(defaultPassword, 10);
     await pool.query(
       'INSERT INTO admin_users (username, password_hash) VALUES ($1, $2)',
       [defaultUsername, passwordHash]
     );

     logger.warn(`Default admin user created: ${defaultUsername} / ${defaultPassword} - CHANGE PASSWORD!`);
   }
   ```
2. Call `seedAdminUser()` in `src/index.ts` after database connection
3. Document in README: Default credentials are `admin` / `changeme`

**Verification**:
- First run creates admin user with hashed password
- Subsequent runs skip creation (idempotent)
- Can login with default credentials
- Warning logged on startup
